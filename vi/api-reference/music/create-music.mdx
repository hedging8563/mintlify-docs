Tạo nhạc và lời bài hát bằng AI. Đây là một API bất đồng bộ. Phản hồi tạo trả về danh tính tác vụ và, khi có sẵn, cũng có thể bao gồm `poll_url` ưu tiên.

## Nội dung yêu cầu

<ParamField body="model" type="string" default="suno_music">
  Model để sử dụng: `suno_music` để tạo nhạc, `suno_lyrics` chỉ để tạo lời bài hát.
</ParamField>

<ParamField body="prompt" type="string" required>
  Mô tả về bản nhạc cần tạo. Có thể bao gồm phong cách, tâm trạng, nhạc cụ, v.v.
</ParamField>

<ParamField body="title" type="string">
  Tiêu đề cho bài hát được tạo.
</ParamField>

<ParamField body="tags" type="string">
  Các thẻ phong cách (ví dụ: "pop, upbeat, electronic").
</ParamField>

<ParamField body="action" type="string">
  Loại tạo: `MUSIC` (mặc định) hoặc `LYRICS`.
</ParamField>

<ParamField body="mv" type="string">
  Phiên bản model để sử dụng.
</ParamField>

<ParamField body="continue_clip_id" type="string">
  ID của một clip trước đó để tiếp tục.
</ParamField>

<ParamField body="continue_at" type="number">
  Dấu thời gian (tính bằng giây) để tiếp tục.
</ParamField>

<ParamField body="user" type="string">
  Một mã định danh duy nhất cho người dùng cuối.
</ParamField>

## Phản hồi

<ResponseField name="id" type="string">
  ID tác vụ để kiểm tra trạng thái.
</ResponseField>

<ResponseField name="task_id" type="string">
  Bí danh của mã tác vụ bất đồng bộ khi được adapter trả về.
</ResponseField>

<ResponseField name="poll_url" type="string">
  URL polling ưu tiên khi có sẵn.
</ResponseField>

<ResponseField name="status" type="string">
  Trạng thái tác vụ: `pending`, `processing`, `completed`, hoặc `failed`.
</ResponseField>

<ResponseField name="created" type="integer">
  Dấu thời gian Unix khi tạo tác vụ.
</ResponseField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    payload := map[string]interface{}{
        "model":  "suno_music",
        "prompt": "An upbeat electronic dance track with heavy bass",
        "title":  "Night Drive",
        "tags":   "electronic, EDM, energetic",
    }
    body, _ := json.Marshal(payload)

    req, _ := http.NewRequest("POST", "https://api.aisonar.dev/v1/music/generations", bytes.NewBuffer(body))
    req.Header.Set("Authorization", "Bearer sk-your-api-key")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, _ := client.Do(req)
    defer resp.Body.Close()

    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Printf("Task ID: %s\n", result["id"])
}
```
